-- Functionality for the Submit Key Button
SubmitKeyButton.MouseButton1Click:Connect(function()
    if TextBox.Text == CorrectKey then
        StatusLabel.Text = "Ready"
        StatusLabel.TextColor3 = Color3.fromRGB(0, 255, 0) -- Change text color to green
        wait(1) -- Short delay for visual feedback
        ScreenGui:Destroy() -- Close the key input UI
        
        -- Load the actual script after successful key entry
        loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/dawid-scripts/Fluent/master/Addons/SaveManager.lua"))()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/dawid-scripts/Fluent/master/Addons/InterfaceManager.lua"))()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/taomlkkk/test/refs/heads/main/taidz%20hub%20(1).txt"))()
        
    else
        StatusLabel.Text = "Invalid"
        StatusLabel.TextColor3 = Color3.fromRGB(255, 0, 0) -- Change text color to red
    end
end)

-- Close UI Button Functionality
CloseButton.MouseButton1Click:Connect(function()
